    Function letterCount

    Returns an associative array of each character, and the number of times it appears.

    auto auto letterCount(Range) (
      Range rng,
      CountMode mode = CountMode.alphaOnly
    );

    Parameters

    NameDescription
    rng A range of characters
    mode Which characters to count

    Example

    assert (letterCount ("hello world 123").countOf("l")==3); // default mode is CountMode.alphaOnly
assert (letterCount ("hello world 123").countOf("2")==0);
assert (letterCount ("hello world 123",CountMode.alphaNumeric).countOf("2")==1);
assert (letterCount ("hello world 123",CountMode.alphaNumeric).countOf(" ")==0);
assert (letterCount ("hello world 123",CountMode.all).countOf(" ")==2);
}

enum CountMode {alphaOnly, alphaNumeric, all
